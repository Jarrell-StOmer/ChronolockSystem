#Import necessary libraries
import random
from datetime import datetime, timedelta

#Define the main function and return the passcode, created time, and expiration time values
#Can add the code to get these values into the database later
def main():
    passcode, created_time, expired_at = generate_passcode()
    print(f"The passcode is: {passcode}")
    print(f"The passcode was created at: {created_time}")
    print(f"The passcode will expire at: {expired_at}")
    
#Generate random integer for passcode and log created time, and calculate expiration time
def generate_passcode():
    passcode = random.randint(1000, 9999)
    created_time = datetime.now()
    duration = timedelta(days=0, hours = 0, minutes = 5, seconds = 0)
    expired_at = created_time + duration
    
    return passcode, created_time, expired_at

main()
